%% Note: Depending on the Mermaid version, UML Use Case diagrams are either partially supported 
        %% or modeled using flowcharts. The following syntax is a popular adaptation.
        
        flowchart LR
            %% Actors
            Pharmacist((Pharmacist))
            Admin((System Admin))
            Supplier((Supplier))
            Patient((Patient))

            %% System Boundary
            subgraph RxNimbus
                UC1([Prescription Validation])
                UC2([Stock Reconciliation])
                UC3([Process Billing])
                UC4([Update Inventory])
                UC5([User Role Management])
                UC6([Automated Restock Requests])
                UC7([Manual Stock Correction])
                UC8([View Medical History])
                UC9([Receive Restock Order])
            end

            %% Actor to Use Case Relationships
            Pharmacist --> UC1
            Pharmacist --> UC2
            Pharmacist --> UC3
            Pharmacist --> UC8

            Admin --> UC5
            Admin --> UC2

            Supplier --> UC9

            Patient --> UC3
            Patient --> UC8

            %% Use Case Relationships
            UC3 -. "<>" .-> UC4
            UC1 -. "<>" .-> UC2
            UC6 -. "<>" .-> UC9
            UC2 -. "<>" .-> UC7
            UC2 -. "<> (low stock)" .-> UC6
